168 message += (monWebServeur.method() == HTTP_GET) ? "GET" : "POST";
169 message += "\nArguments: ";
170 message += monWebServeur.args();
171 message += "\n";
172 for (uint8_t i = 0; i < monWebServeur.args(); i++) {

176 // Envoi de la réponse HTML

183void setupWebServer(){
184 // On a besoin d'une connexion WiFi !
185 // Test de la connexion WiFi. Si elle n'est pas active alors on démarre le WiFi
186 if (WiFi.status() != WL_CONNECTED){setupWiFi();} // Connexion WiFi
187 MYDEBUG_PRINTLN("-WEBSERVER : Démarrage");
188
189 // Configuration de mon serveur web en définissant plusieurs routes
190 // A chaque route est associée une fonction
191 monWebServeur.on("/", handleRoot);
192 monWebServeur.on("/scan", handleScan);
193 monWebServeur.on("/config", handleConfig);
194 monWebServeur.on("/adafruit", handleAdafruit);
195 monWebServeur.onNotFound(handleNotFound);
196// monWebServeur.on("/format", handleFormat); // A ajouter quand le SPIFFFS est activé
197
198 monWebServeur.begin(); // Démarrage du serveur
199 MYDEBUG_PRINTLN("-WEBSERVER : Serveur Web démarré");
200}
