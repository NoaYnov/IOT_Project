435void setupWebServer(){
436 // On a besoin d'une connexion WiFi !
437 // Test de la connexion WiFi. Si elle n'est pas active alors on démarre le WiFi
438 if (WiFi.status() != WL_CONNECTED){setupWiFi();} // Connexion WiFi
439 MYDEBUG_PRINTLN("-WEBSERVER : Démarrage");
440
441 // Configuration de mon serveur web en définissant plusieurs routes
442 // A chaque route est associée une fonction
443 monWebServeur.on("/", handleRoot);
444 monWebServeur.on("/scan", handleScan);
445 monWebServeur.on("/config", handleConfig);
446 monWebServeur.on("/adafruit", handleAdafruit);
447 monWebServeur.on("/contact_tracer", handleContactTracer);
448 monWebServeur.onNotFound(handleNotFound);
449 monWebServeur.on("/format", handleFormat); // A ajouter quand le SPIFFFS est activé
